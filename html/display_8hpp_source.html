52 static inline void write_byte(uint_fast8_t byte, Pio* port, uint_fast32_t mask, uint32_t & next_time) __attribute__((always_inline, unused));
53 static inline void write_byte(uint_fast8_t byte, Pio* port, uint_fast32_t mask, uint32_t & next_time){
54 
55  for(uint_fast8_t i = 0; i < 8; i++){
56  while(DUE_TIMER_VAL < next_time);
57  next_time = DUE_TIMER_VAL + 105; // 1,25 us * 84MHz = 105 // 1,25 us is the time of a complete bit writing (amount of cycles)
58  port -> PIO_SODR = mask;
59 
60  if(byte & 0b10000000){ //check MSB
61 
62  while( (next_time - DUE_TIMER_VAL) > 31 ); // Timing from FastLED's library // Write a 1
63 
64  }
65  else{
66  while( (next_time - DUE_TIMER_VAL) > 84 ); // Timing from FastLED's library // write a 0
67 
68  }
69  port -> PIO_CODR = mask;
70  byte = byte << 1;
71  }
72 
73 }

101  void set_led(unsigned int x, unsigned int y, uint32_t color){
102  size_t index;
103  if(y%2==0){
104  index = y*columns + x;
105  }
106  else{
107  index = (y+1)*columns - 1 - x;
108  }
109  leds[index] = color;
110  }
113  void update(){
114  tc_init(DUE_TIMER,DUE_TIMER_CHANNEL,TC_CMR_TCCLKS_TIMER_CLOCK1);
115  pmc_enable_periph_clk(DUE_TIMER_ID);
116  tc_start(DUE_TIMER,DUE_TIMER_CHANNEL);
117  port -> PIO_CODR = mask;
118  __disable_irq(); __disable_fault_irq();
119 
120  uint32_t next_time = DUE_TIMER_VAL + 105;
121 
122  for(size_t i = 0; i < rows*columns; i++){
123  write_byte(((leds[i] >> 8) & 0xFF), port, mask, next_time);
124  write_byte(((leds[i] >> 16) & 0xFF), port, mask, next_time);
125  write_byte(((leds[i]) & 0xFF), port, mask, next_time);
126  }
127 
128  __enable_irq(); __enable_fault_irq();
129  tc_stop(DUE_TIMER,DUE_TIMER_CHANNEL);
130 
131  }
